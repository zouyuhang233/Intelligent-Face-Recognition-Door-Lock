## 1. 硬件连接

```
MCU STM32F103ZET6          摄像头 FM225
  PA2 (USART2_TX)    ────  UART_RX
  PA3 (USART2_RX)    ────  UART_TX
  VCC                ────  5.5V~9V ≥1A

MCU                       PC 调试
  PB10 (USART3_TX)  ────  串口 RX (115200,8N1)
  PB11 (USART3_RX)  ────  串口 TX
```

| 串口 | 引脚 | 波特率 | 用途 |
|------|------|--------|------|
| USART2 | PA2(TX) PA3(RX) | 115200 | 摄像头通信 |
| USART3 | PB10(TX) PB11(RX) | 115200 | PC 调试输出 |

#include "fm22x.h"
#include "usart.h"
#include <string.h>
#include "lcd.h"

uint8_t fm22x_rx_byte = 0;

/* ---- 接收状态机 ---- */
static uint8_t rx_state = 0;
static uint8_t rx_parity = 0;
static uint8_t rx_msgid = 0;
static uint16_t rx_size = 0;
static uint16_t rx_index = 0;
static uint16_t rx_data_idx = 0;
static uint8_t rx_buffer[RX_BUFFER_SIZE];
static uint8_t rx_data_buffer[1024];

volatile bool module_ready = false;
volatile bool command_complete = false;
volatile uint8_t last_result = 0xFF;

#define STATE_WAIT_SYNC0    0
#define STATE_WAIT_SYNC1    1
#define STATE_WAIT_MSGID    2
#define STATE_WAIT_SIZE_H   3
#define STATE_WAIT_SIZE_L   4
#define STATE_WAIT_DATA     5
#define STATE_WAIT_PARITY   6

/* ---- 校验 ---- */
uint8_t FM22x_CalculateParity(uint8_t* data, uint16_t len)
{
    uint8_t parity = 0;
    for(uint16_t i = 0; i < len; i++)
        parity ^= data[i];
    return parity;
}

/* ---- 发送 ---- */
void FM22x_SendCommand(uint8_t* data, uint16_t len)
{
    HAL_UART_Transmit(&huart2, data, len, 100);
}

/* ---- 命令封装 ---- */
void FM22x_Reset(void)
{
    uint8_t cmd[] = {0xEF, 0xAA, MID_RESET, 0x00, 0x00, 0x00};
    cmd[5] = FM22x_CalculateParity(&cmd[2], 3);
    FM22x_SendCommand(cmd, sizeof(cmd));
}

void FM22x_GetStatus(void)
{
    uint8_t cmd[] = {0xEF, 0xAA, MID_GETSTATUS, 0x00, 0x00, 0x00};
    cmd[5] = FM22x_CalculateParity(&cmd[2], 3);
    FM22x_SendCommand(cmd, sizeof(cmd));
}

void FM22x_Verify(uint8_t pd_rightaway, uint8_t timeout)
{
    uint8_t cmd[] = {0xEF, 0xAA, MID_VERIFY, 0x00, 0x02,
                     pd_rightaway, timeout, 0x00};
    cmd[7] = FM22x_CalculateParity(&cmd[2], 5);
    FM22x_SendCommand(cmd, sizeof(cmd));
}

void FM22x_EnrollSingle(uint8_t admin, uint8_t face_dir, uint8_t timeout)
{
    uint8_t cmd[6 + USER_NAME_SIZE + 2];
    uint16_t idx = 0;
    cmd[idx++] = 0xEF; cmd[idx++] = 0xAA;
    cmd[idx++] = MID_ENROLL_SINGLE;
    cmd[idx++] = 0x00; cmd[idx++] = 0x00;
    uint16_t data_size = 1 + USER_NAME_SIZE + 2;
    cmd[3] = (data_size >> 8) & 0xFF;
    cmd[4] = data_size & 0xFF;
    cmd[idx++] = admin;
    for(uint8_t i = 0; i < USER_NAME_SIZE; i++) cmd[idx++] = 0x00;
    cmd[idx++] = face_dir;
    cmd[idx++] = timeout;
    uint8_t parity = FM22x_CalculateParity(&cmd[2], idx - 2);
    cmd[idx++] = parity;
    FM22x_SendCommand(cmd, idx);
}

void FM22x_Enroll(uint8_t admin, uint8_t face_dir, uint8_t timeout)
{
    uint8_t cmd[6 + USER_NAME_SIZE + 2];
    uint16_t idx = 0;
    cmd[idx++] = 0xEF; cmd[idx++] = 0xAA;

/* ---- 全局人脸状态 ---- */
volatile int16_t g_face_left, g_face_top, g_face_right, g_face_bottom;
volatile int16_t g_face_yaw, g_face_pitch, g_face_roll;
volatile int16_t g_face_state;
volatile bool g_face_updated = false;
volatile uint32_t g_face_count = 0;
volatile uint16_t g_last_uid = 0;
volatile uint8_t g_user_count = 0;

/* ---- 解析人脸状态 ★ 关键字段序 ---- */
static void ParseFaceState(uint8_t* data, uint16_t size)
{
    if(size < 17) return;

    /* 小端序。字段: left top right bottom yaw pitch roll state spare */
    int16_t left   = (int16_t)(data[1] | (data[2] << 8));
    int16_t top    = (int16_t)(data[3] | (data[4] << 8));
    int16_t right  = (int16_t)(data[5] | (data[6] << 8));
    int16_t bottom = (int16_t)(data[7] | (data[8] << 8));
    int16_t yaw    = (int16_t)(data[9] | (data[10] << 8));
    int16_t pitch  = (int16_t)(data[11] | (data[12] << 8));
    int16_t roll   = (int16_t)(data[13] | (data[14] << 8));
    int16_t state  = data[15];

    g_face_left   = left;
    g_face_top    = top;
    g_face_right  = right;
    g_face_bottom = bottom;
    g_face_yaw    = yaw;
    g_face_pitch  = pitch;
    g_face_roll   = roll;
    g_face_state  = state;
    g_face_updated = true;
    g_face_count++;
}

uint8_t lock_state = 0;
uint8_t beep_state = 0;

static void ParseResult(uint8_t result)
{
    if (result == MR_SUCCESS)
        lock_state = 1;
    else if (result == MR_FAILED4_TIMEOUT)
        beep_state = 1;
}

uint8_t lock_beep_state = 0;

/* ---- 处理完整帧（中断上下文调用） ---- */
static void ProcessReceivedFrame(uint8_t msgid, uint16_t size, uint8_t* data)
{
    switch(msgid)
    {
        case MID_REPLY:   // 0x00 命令回复
            if(size >= 2) {
                uint8_t cmd = data[0];
                uint8_t result = data[1];
                last_result = result;
                command_complete = true;
                if(cmd == MID_VERIFY) lock_beep_state = 1;
                ParseResult(result);
                if((cmd == MID_ENROLL || cmd == MID_ENROLL_SINGLE
                    || cmd == MID_VERIFY)
                    && result == MR_SUCCESS && size >= 4)
                    g_last_uid = ((uint16_t)data[2] << 8) | data[3];
                if(cmd == 0x24 && result == MR_SUCCESS && size >= 3)
                    g_user_count = data[2];
            }
            break;

        case MID_NOTE:    // 0x01 主动通知
            if(size >= 1) {
                uint8_t nid = data[0];
                if(nid == 0x00)      module_ready = true;
                else if(nid == 0x01) ParseFaceState(data, size);
            }
            break;

        default: break;
    }
}

/* ---- 初始化 ---- */
void FM22x_Init(void)
{
    module_ready = false;
    command_complete = false;
    last_result = 0xFF;
    rx_state = STATE_WAIT_SYNC0;
    rx_index = 0;
    rx_data_idx = 0;
    rx_size = 0;
}

/* ---- UART 接收状态机（中断调用） ---- */
void FM22x_UART_Callback(uint8_t byte)
{
    switch(rx_state)
    {
        case STATE_WAIT_SYNC0:
            if(byte == 0xEF) {
                rx_state = STATE_WAIT_SYNC1;
                rx_parity = 0;
                rx_index = 0;
                rx_data_idx = 0;
                rx_buffer[rx_index++] = byte;
            }
            break;

        case STATE_WAIT_SYNC1:
            if(byte == 0xAA) {
                rx_state = STATE_WAIT_MSGID;
                rx_buffer[rx_index++] = byte;
            } else {
                rx_state = STATE_WAIT_SYNC0;
                rx_parity = 0;
                rx_index = 0;
                rx_data_idx = 0;
            }
            break;

        case STATE_WAIT_MSGID:
            rx_msgid = byte;
            rx_parity ^= byte;
            rx_buffer[rx_index++] = byte;
            rx_state = STATE_WAIT_SIZE_H;
            break;

        case STATE_WAIT_SIZE_H:
            rx_size = ((uint16_t)byte << 8);
            rx_parity ^= byte;
            rx_buffer[rx_index++] = byte;
            rx_state = STATE_WAIT_SIZE_L;
            break;

        case STATE_WAIT_SIZE_L:
            rx_size |= byte;
            rx_parity ^= byte;
            rx_buffer[rx_index++] = byte;
            rx_state = (rx_size == 0) ? STATE_WAIT_PARITY : STATE_WAIT_DATA;
            if(rx_size > 0) rx_data_idx = 0;
            break;

        case STATE_WAIT_DATA:
            rx_data_buffer[rx_data_idx++] = byte;
            rx_parity ^= byte;
            rx_buffer[rx_index++] = byte;
            if(rx_data_idx >= rx_size) rx_state = STATE_WAIT_PARITY;
            break;

        case STATE_WAIT_PARITY:
            rx_buffer[rx_index++] = byte;
            if(byte == rx_parity)
                ProcessReceivedFrame(rx_msgid, rx_size, rx_data_buffer);
            rx_state = STATE_WAIT_SYNC0;
            break;

        default:
            rx_state = STATE_WAIT_SYNC0;
            break;
    }
}

static void uart3_printf(const char *fmt, ...)
{
    char buf[256];
    va_list args;
    va_start(args, fmt);
    vsnprintf(buf, sizeof(buf), fmt, args);
    va_end(args);
    HAL_UART_Transmit(&huart3, (uint8_t *)buf, strlen(buf), 500);
}

static void lcd_line(uint16_t y, const char *text, uint16_t color)
{
    LCD_FillRect(10, y, 310, y + 20, WHITE);
    LCD_ShowString(10, y, (char *)text, color, WHITE);
}

static void led_blink(uint8_t n, uint8_t fast)
{
    uint32_t d = fast ? 100 : 200;
    for (uint8_t i = 0; i < n; i++) {
        HAL_GPIO_WritePin(GPIOE, GPIO_PIN_5, GPIO_PIN_RESET);
        HAL_Delay(d);
        HAL_GPIO_WritePin(GPIOE, GPIO_PIN_5, GPIO_PIN_SET);
        HAL_Delay(d);
    }
}

/* ---- 主函数 ---- */
int main(void)
{
    HAL_Init();
    SystemClock_Config();

    MX_GPIO_Init();
    MX_USART2_UART_Init();
    MX_USART3_UART_Init();
    MX_FSMC_Init();

    /* 初始化 */
    LCD_Init();
    LCD_Clear(WHITE);
    LCD_ShowString(10, 5, "=== Face Recognition Test ===", BLUE, WHITE);
    uart3_printf("\r\n==== Face Recognition System ====\r\n");

    HAL_GPIO_WritePin(LCD_BL_GPIO_Port, LCD_BL_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_3, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_5, GPIO_PIN_SET);
    HAL_Delay(500);
    pa0_last = HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0);
    pe3_last = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_3);
    pe2_last = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_2);

    /* 摄像头初始化 */
    FM22x_Init();
    HAL_UART_Receive_IT(&huart2, &fm22x_rx_byte, 1);
    lcd_line(30, "Waiting camera...", BLACK);

    {
        uint32_t t0 = HAL_GetTick();
        while ((HAL_GetTick() - t0) < 15000) {
            if (module_ready) {
                lcd_line(30, "Camera READY!", GREEN);
                uart3_printf("[INIT] Camera READY!\r\n");
                led_blink(2, 1);
                break;
            }
            HAL_Delay(10);
        }
    }
    if (!module_ready) {
        lcd_line(30, "Camera FAIL!", RED);
        uart3_printf("[ERROR] Camera not ready!\r\n");
    }

    lcd_line(320, "=== Controls ===", BLUE);
    lcd_line(340, "PE3: Enroll", GRAY);
    lcd_line(360, "PE2: Verify", GRAY);
    lcd_line(380, "PA0: Delete All", GRAY);
    uart3_printf("[INFO] PE3=Enroll PE2=Verify PA0=DeleteAll\r\n\r\n");

    /* ---- 主循环 ---- */
    while (1)
    {
        /* 人脸状态显示 */
        if (g_face_updated) {
            g_face_updated = false;
            char buf[60];
            int16_t yaw_deg = g_face_yaw / 10;  // 0.1°→°

            uart3_printf("[FACE] #%lu state=%d box=(%d,%d,%d,%d) yaw=%d\r\n",
                         (unsigned long)g_face_count, (int)g_face_state,
                         g_face_left, g_face_top, g_face_right, g_face_bottom,
                         yaw_deg);

            sprintf(buf, "Face: %s",
                    g_face_state == 0 ? "DETECTED" :
                    g_face_state == 1 ? "NO FACE" : "???");
            lcd_line(80, buf, (g_face_state == 0) ? GREEN : RED);

            sprintf(buf, "Yaw:%+3d Box:(%d,%d,%d,%d)",
                    yaw_deg, g_face_left, g_face_top,
                    g_face_right, g_face_bottom);
            lcd_line(100, buf, BLACK);

            if (enrolling) {
                sprintf(buf, "Enrolling... #%lu", (unsigned long)g_face_count);
                lcd_line(160, buf, BLUE);
            }
        }

        /* 命令结果处理 */
        if (command_complete) {
            command_complete = false;
            uint8_t r = last_result;

            if (enrolling) {
                enrolling = 0;
                if (r == 0) {
                    lcd_line(160, "Enroll OK!", GREEN);
                    uart3_printf("[ENROLL] SUCCESS!\r\n");
                    led_blink(3, 0);
                    enroll_ok_count++;
                } else {
                    char b[40]; sprintf(b, "Enroll FAIL: %d", r);
                    lcd_line(160, b, RED);
                    uart3_printf("[ENROLL] FAILED: code=%d\r\n", r);
                    led_blink(1, 1);
                    enroll_fail_count++;
                }
            }
            else if (verifying) {
                verifying = 0;
                if (r == 0) {
                    lcd_line(160, "Verify OK!", GREEN);
                    uart3_printf("[VERIFY] SUCCESS!\r\n");
                    led_blink(3, 0);
                    verify_ok_count++;
                    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_3, GPIO_PIN_SET);
                    lcd_line(300, "LOCK: OPEN ", GREEN);
                } else {
                    char b[40]; sprintf(b, "Verify FAIL: %d", r);
                    lcd_line(160, b, RED);
                    uart3_printf("[VERIFY] FAILED: code=%d\r\n", r);
                    led_blink(1, 1);
                    verify_fail_count++;
                    lcd_line(300, "LOCK: CLOSE", RED);
                }
            }
        }

        /* 自动关锁 */
        {
            extern uint8_t lock_state;
            static uint32_t lock_tick = 0;
            if (lock_state) {
                if (lock_tick == 0) lock_tick = HAL_GetTick();
                if (HAL_GetTick() - lock_tick > 3000) {
                    lock_state = 0; lock_tick = 0;
                    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_3, GPIO_PIN_RESET);
                    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_5, GPIO_PIN_SET);
                    lcd_line(300, "LOCK: CLOSE", RED);
                    uart3_printf("[LOCK] Auto close\r\n");
                }
            } else { lock_tick = 0; }
        }

        /* 按键 */
        uint8_t key = 0;
        uint8_t pa0 = HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0);
        if (pa0 == 1 && pa0_last == 0) key = 1; pa0_last = pa0;
        uint8_t pe3 = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_3);
        if (pe3 == 0 && pe3_last == 1) key = 2; pe3_last = pe3;
        uint8_t pe2 = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_2);
        if (pe2 == 0 && pe2_last == 1) key = 3; pe2_last = pe2;

        if (key && (HAL_GetTick() - key_tick > 300)) {
            key_tick = HAL_GetTick();
            if (key == 1) {      // PA0: 删除全部
                uart3_printf("\r\n[KEY] PA0 -> Delete All\r\n");
                lcd_line(140, "Deleting all...", RED);
                FM22x_DeleteAll();
            }
            else if (key == 2) { // PE3: 录入
                uart3_printf("\r\n[KEY] PE3 -> Enroll\r\n");
                enrolling = 1;
                lcd_line(160, "Enrolling...", BLUE);
                lcd_line(180, "Turn head slowly U-D-L-R", YELLOW);
                FM22x_Enroll(0, 0x1F, 30);
            }
            else if (key == 3) { // PE2: 验证
                uart3_printf("\r\n[KEY] PE2 -> Verify\r\n");
                verifying = 1;
                lcd_line(160, "Verifying...", BLUE);
                lcd_line(180, "Look at camera!", YELLOW);
                FM22x_Verify(1, 10);
            }
        }
        HAL_Delay(100);
    }
}

## 5. 数据流架构

```
摄像头 UART_TX
    │ 115200bps
    ▼
MCU PA3 (USART2_RX) ──→ USART2_IRQHandler
    │                    └→ HAL_UART_IRQHandler
    │                         └→ HAL_UART_RxCpltCallback (esp8266.c)
    │                              │
    │                       USART2 分支:
    │                       FM22x_UART_Callback(byte)  ← 逐字节状态机
    │                         │
    │                         ├─ 收集 EF AA MsgID Size Data Checksum
    │                         └─ 校验通过 → ProcessReceivedFrame()
    │                              │
    │                              ├─ REPLY: command_complete=true
    │                              │         last_result=结果码
    │                              │
    │                              └─ NOTE(NID_FACE_STATE):
    │                                   ParseFaceState()
    │                                   g_face_updated=true
    │                                   g_face_*=坐标/姿态/状态
    │
    ▼
main() 循环 (100ms/次)
    │
    ├─ g_face_updated? → LCD显示 + USART3输出
    ├─ command_complete + enrolling?  → 录入结果
    ├─ command_complete + verifying?  → 验证结果+开锁
    ├─ 按键检测 → FM22x_Enroll/Verify/DeleteAll
    └─ 自动关锁(3s)
```

## 6. 协议帧格式

```
EF AA | MsgID | SizeH SizeL | Data[0..N] | Checksum
 2B   |  1B   |  2B(大端)   |   N字节     |  1B

校验: Checksum = MsgID ^ SizeH ^ SizeL ^ Data[0] ^ ... ^ Data[N-1]
```

## 7. NID_FACE_STATE 字段序 ★★★

**这是整个项目最关键的发现。官方开发手册写的字段序与实际固件不一致。**

```
实际固件字节序 (已验证数据自洽):

data[ 0] = NID (固定 0x01)
data[ 1.. 2] = left   (int16, LE)  ← 框左边
data[ 3.. 4] = top    (int16, LE)  ← 框上边
data[ 5.. 6] = right  (int16, LE)  ← 框右边
data[ 7.. 8] = bottom (int16, LE)  ← 框下边
data[ 9..10] = yaw    (int16, LE)  ← 偏航(0.1°)
data[11..12] = pitch  (int16, LE)  ← 俯仰
data[13..14] = roll   (int16, LE)  ← 翻滚
data[15]     = state  (uint8)      ← 人脸状态 0=检测到
data[16]     = spare  (uint8)      ← 保留
```

## 8. 操作说明

### 录入 (PE3)
1. 正对摄像头 30-50cm
2. 按 PE3，LCD 显示 `Enrolling...`
3. **缓慢大幅度转头**：上→下→左→右，每个方向停 2 秒
4. LCD 显示 `Enroll OK!` = 成功
5. 30 秒内完成

### 验证 (PE2)
1. 正对摄像头
2. 按 PE2，LCD 显示 `Verifying...`
3. `Verify OK!` = 识别成功，开锁 3 秒
4. `Verify FAIL: 8` = 无匹配人脸（未录入）

### 删除 (PA0)
1. 按 PA0，所有用户被删除
